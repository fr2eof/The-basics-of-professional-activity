ls просмотр директории -la всех файлов, даже скрытых
mkdir name_dir создание директории
touch name_file создание файла
cd name_dir смена директории
cat name_file прочитать файл
cat >file_name поотоковый ввод данных в файл
chmod new_root (600) name_file  смена прав доступа(user group )
chmod +- g+-r name_file добавить права
pwd возвращает путь директории, в которой находимся
pwd >file_name перезаписывает внутри файла
pwd >>file_name дозаписать в файл
vi/vim name_file

ln -s name_first name_second символическая ссылка
